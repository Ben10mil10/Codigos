using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Programacion_Semana_9
{
    internal class Program
    { //Fritz Sebastian Rockville Torres ID:1125705
        public double AreaCuadrado(double lado)
        {
            return lado * lado;
        }

        public double AreaRectangulo(double bace, double altura)
        {
            return bace * altura;
        }

        public double AreaTriangulo(double bace, double altura)
        {
            return (bace * altura) / 2;
        }

        public double AreaCono(double radio, double altura)
        {
            double pi = Math.PI;
            return pi * radio * (radio + Math.Sqrt(altura * altura + radio * radio));
        }

        public double HipotenusaPitagoras(double cateto1, double cateto2)
        {
            return Math.Sqrt(cateto1 * cateto1 + cateto2 * cateto2);
        }

        public double AreaCirculo(double radio)
        {
            double pi = Math.PI;
            return pi * radio * radio;
        }

        public static void Main(string[] args)
        {
            Program programa = new Program();
            bool continuar = true;

            while (continuar)
            {
                Console.WriteLine("Seleccione la operación geométrica que desea realizar:");
                Console.WriteLine("1. Área del cuadrado");
                Console.WriteLine("2. Área del rectángulo");
                Console.WriteLine("3. Área del triángulo");
                Console.WriteLine("4. Área del círculo");
                Console.WriteLine("5. Área del cono");
                Console.WriteLine("6. Hipotenusa (Teorema de Pitágoras)");
                Console.WriteLine("7. Salir");

                int opcion = int.Parse(Console.ReadLine());

                switch (opcion)
                {
                    case 1:
                        Console.Write("Ingrese el lado del cuadrado: ");
                        double lado = double.Parse(Console.ReadLine());
                        Console.WriteLine("El área del cuadrado es: " + programa.AreaCuadrado(lado));
                        break;

                    case 2:
                        Console.Write("Ingrese la base del rectángulo: ");
                        double baseRect = double.Parse(Console.ReadLine());
                        Console.Write("Ingrese la altura del rectángulo: ");
                        double alturaRect = double.Parse(Console.ReadLine());
                        Console.WriteLine("El área del rectángulo es: " + programa.AreaRectangulo(baseRect, alturaRect));
                        break;

                    case 3:
                        Console.Write("Ingrese la base del triángulo: ");
                        double baseTri = double.Parse(Console.ReadLine());
                        Console.Write("Ingrese la altura del triángulo: ");
                        double alturaTri = double.Parse(Console.ReadLine());
                        Console.WriteLine("El área del triángulo es: " + programa.AreaTriangulo(baseTri, alturaTri));
                        break;

                    case 4:
                        Console.Write("Ingrese el radio del círculo: ");
                        double radioCirc = double.Parse(Console.ReadLine());
                        Console.WriteLine("El área del círculo es: " + programa.AreaCirculo(radioCirc));
                        break;

                    case 5:
                        Console.Write("Ingrese el radio del cono: ");
                        double radioCono = double.Parse(Console.ReadLine());
                        Console.Write("Ingrese la altura del cono: ");
                        double alturaCono = double.Parse(Console.ReadLine());
                        Console.WriteLine("El área del cono es: " + programa.AreaCono(radioCono, alturaCono));
                        break;

                    case 6:
                        Console.Write("Ingrese el primer cateto: ");
                        double cateto1 = double.Parse(Console.ReadLine());
                        Console.Write("Ingrese el segundo cateto: ");
                        double cateto2 = double.Parse(Console.ReadLine());
                        Console.WriteLine("La hipotenusa es: " + programa.HipotenusaPitagoras(cateto1, cateto2));
                        break;

                    case 7:
                        continuar = false;
                        Console.WriteLine("Saliendo...");
                        break;

                    default:
                        Console.WriteLine("Opción no válida, intente nuevamente.");
                        break;
                }

                Console.WriteLine();
            }
        }
    }
}
